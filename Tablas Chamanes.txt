CREATE TABLE PROYECTO
(CODIGO_PROYECTO VARCHAR(8) CONSTRAINT PRO_COD_PRO_PK PRIMARY KEY,  
NOMBRE VARCHAR(40) CONSTRAINT PRO_NOM_NN NOT NULL,
DIRECCION VARCHAR(20) CONSTRAINT PROO_DIR_NN NOT NULL);

INSERT INTO PROYECTO 
VALUES ('&CODIGO_PROYECTO', '&NOMBRE', '&DIRECCION');

CREATE TABLE TRABAJADOR 
(CED_TRABAJADOR NUMBER(8) CONSTRAINT TRA_CED_TRA_PK PRIMARY KEY,
 NOMBRE_COMPLETO VARCHAR(40) CONSTRAINT TRA_NOMBRE_COMPLETO_NN NOT NULL,
 CELULAR NUMBER(10) CONSTRAINT ASESOR_CELULAR_NN NOT NULL,
 CORREO_ELECTRONICO VARCHAR(20) CONSTRAINT TRA_COR_ELE_NN NOT NULL,
 DIRECCION_RESIDENCIA VARCHAR(20),  
 CONTRASENA VARCHAR(10) CONSTRAINT TRA_CON_NN NOT NULL,  
 ESTADO VARCHAR2(10) CONSTRAINT TRA_EST_NN NOT NULL,
 ID_ROL NUMBER(8) CONSTRAINT TRA_ID_ROL_NN NOT NULL,
 CONSTRAINT TRA_EST_CH CHECK (ESTADO IN ('Activo', 'Inactivo')),
 CONSTRAINT TRA_ID_ROL_FK FOREIGN KEY (ID) REFERENCES ROL(ID));

INSERT INTO TRABAJADOR 
VALUES ('&CED_TRABAJADOR', '&NOMBRE_COMPLETO', '&CELULAR', '&CORREO_ELECTRONICO', '&DIRECCION_RESIDENCIA', '&CONTRASENA', '&ESTADO');

 NOMBRE_ROL VARCHAR(20) CONSTRAINT ROL_NOM_NN NOT NULL);

CREATE TABLE CLIENTE

  NOMBRE_COMPLETO VARCHAR(40) CONSTRAINT CLIENTE_NOMBRE_COMPLETO_NN NOT NULL,

  );

CREATE TABLE TORRE 
( CODIGO_TORRE VARCHAR(8) CONSTRAINT TOR_COD_TOR_PK PRIMARY KEY,
  FECHA_CONSTRUCCION DATE CONSTRAINT TOR_FEC_CONS_NN NOT NULL,
  CODIGO_PROYECTO VARCHAR(8) CONSTRAINT TOR_COD_PRO_NN NOT NULL,
  CONSTRAINT TOR_COD_PRO_FK FOREIGN KEY (CODIGO_PROYECTO) REFERENCES PROYECTO(CODIGO_PROYECTO));


INSERT INTO TORRE  
VALUES ('&CODIGO_TORRE', TO_DATE('&FECHA_CONSTRUCCION', 'DD-MM-YYYY'), '&CODIGO_PROYECTO');

CREATE TABLE PROPIEDAD
(MATRICULA VARCHAR2(8) CONSTRAINT PROP_MAT_PK PRIMARY KEY,
 VALOR FLOAT(20) CONSTRAINT PROP_VAL_NN NOT NULL,
 TIPO VARCHAR2(20) CONSTRAINT PROP_TIP_NN NOT NULL,  
 AREA FLOAT(5) CONSTRAINT PROP_ARE_NN NOT NULL,
 ESTADO VARCHAR2(10) CONSTRAINT PROP_EST_NN NOT NULL,
 PISO NUMBER(4) CONSTRAINT PROP_PIS_NN NOT NULL,
 CODIGO_TORRE VARCHAR2(8) CONSTRAINT PROP_COD_TOR_NN NOT NULL,
 CONSTRAINT PROP_TIP_CHK CHECK (TIPO IN ('Apartamento', 'Garaje', 'Local')),
 CONSTRAINT PROP_EST_CHK CHECK (ESTADO IN ('Disponible', 'Vendido')),
 CONSTRAINT PROP_COD_TOR_FK FOREIGN KEY (CODIGO_TORRE) REFERENCES TORRE(CODIGO_TORRE));



INSERT INTO PROPIEDAD
VALUES ('&MATRICULA', &VALOR, '&TIPO', &AREA, '&ESTADO', &PISO, '&CODIGO_TORRE');

CREATE TABLE VENTA
( ID_VENTA NUMBER(8) CONSTRAINT VEN_ID_PK PRIMARY KEY,
  METODO_PAGO VARCHAR2(20) CONSTRAINT VENT_METO_PAGO_NN NOT NULL,
  PRECIO_TOTAL FLOAT(20) CONSTRAINT VEN_PREC_TOT_NN NOT NULL,
  FECHA_VENTA DATE CONSTRAINT VEN_FEC_NN NOT NULL,
  PORCENTAJE_INTERES FLOAT(5),
  CANTIDAD_CUOTAS NUMBER(20),
  VALOR_SUBSIDIO FLOAT(20),
  CONSTRAINT VENT_METO_PAGO_CHK CHECK (METODO_PAGO IN ('Credito', 'Contado')),
  MATRICULA VARCHAR(8) CONSTRAINT VEN_MAT_NN NOT NULL,
  CONSTRAINT VEN_MAT_FK FOREIGN KEY (MATRICULA) REFERENCES PROPIEDAD(MATRICULA),
  CEDULA_CLIENTE NUMBER(8) CONSTRAINT VEN_CED_CLI_NN NOT NULL,
  CONSTRAINT VEN_CED_CLI_FK FOREIGN KEY (CEDULA_CLIENTE) REFERENCES CLIENTE(CEDULA_CLIENTE),
  CED_TRABAJADOR NUMBER(8) CONSTRAINT VEN_CED_TRAB_NN NOT NULL,
  CONSTRAINT VEN_CED_TRAB_FK FOREIGN KEY (CED_TRABAJADOR) REFERENCES TRABAJADOR(CED_TRABAJADOR));


INSERT INTO VENTA 
VALUES ('&ID_VENTA', '&METODO_PAGO', &PRECIO_TOTAL, TO_DATE('&FECHA_VENTA', 'DD-MM-YYYY'), &PORCENTAJE_INTERES, '&CANTIDAD_CUOTAS', '&VALOR_SUBSIDIO', '&MATRICULA', '&CEDULA_CLIENTE', '&CED_TRABAJADOR');

CREATE TABLE DEUDA 
( ID_DEUDA NUMBER(8) CONSTRAINT DEU_ID_DEU_PK PRIMARY KEY,
  VALOR_CUOTA FLOAT(20) CONSTRAINT DEU_VAL_CUO_NN NOT NULL,
  VALOR_DEUDA FLOAT(20) CONSTRAINT DEU_VAL_DEU_NN NOT NULL,
  ID_VENTA NUMBER(8) CONSTRAINT DEU_ID_VENTA_NN NOT NULL,
  CONSTRAINT DEU_ID_VENTA_FK FOREIGN KEY (ID_VENTA) REFERENCES VENTA(ID_VENTA));

INSERT INTO DEUDA  
VALUES ('&ID_DEUDA', &VALOR_CUOTA, &VALOR_DEUDA, '&ID_VENTA');

CREATE TABLE PAGO 
(
  ID_PAGO NUMBER(8) CONSTRAINT PAGO_ID_PAGO_PK PRIMARY KEY,
  TIPO VARCHAR2(20) CONSTRAINT PAGO_TIPO_NN NOT NULL,
  VALOR FLOAT(20) CONSTRAINT PAGO_VALOR_NN NOT NULL,
  FECHA_PAGO DATE CONSTRAINT PAGO_FECHA_NN NOT NULL,
  CED_TRABAJADOR NUMBER(8) CONSTRAINT PAGO_CED_TRABAJADOR_NN NOT NULL,
  ID_DEUDA NUMBER(8) CONSTRAINT PAGO_ID_DEUDA_NN NOT NULL,
  CONSTRAINT PAGO_TIPO_CHK CHECK (TIPO IN ('Cuota', 'Abono', 'Contado')),
  CONSTRAINT PAGO_CED_TRABAJADOR_FK FOREIGN KEY (CED_TRABAJADOR) REFERENCES TRABAJADOR(CED_TRABAJADOR),
  CONSTRAINT PAGO_ID_DEUDA_FK FOREIGN KEY (ID_DEUDA) REFERENCES DEUDA(ID_DEUDA));


INSERT INTO PAGO  
VALUES ('&ID_PAGO', '&TIPO', '&VALOR', TO_DATE('&FECHA_PAGO', 'DD-MM-YYYY'), '&CED_TRABAJADOR', '&ID_DEUDA');
